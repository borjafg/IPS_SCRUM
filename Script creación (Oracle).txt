CREATE TABLE Clientes(id_cliente number(32), nombre varchar(64), metodo_pago varchar(64), direccion varchar(255), PRIMARY KEY(id_cliente));


CREATE TABLE Pedidos(id_pedido number(32), id_cliente number(32), fecha date, estado varchar(64), direccion varchar(255), PRIMARY KEY(id_pedido), FOREIGN KEY(id_cliente) REFERENCES Clientes(id_cliente));


CREATE TABLE PosicionProductos(id_posicion number (32), pasillo number(10), estanteria varchar(32),altura number(10), posicionX number(10), primary key (id_posicion));


CREATE TABLE Productos(id_producto number(32), id_posicion number(32), nombre varchar(64), categoria varchar(64), precio number(10,2), PRIMARY KEY(id_producto), foreign key(id_posicion) references Posicionproductos(id_posicion));


CREATE TABLE ProductosPedido(id_producto number(32), id_pedido number (32), cantidad number(10), primary key(id_producto, id_pedido), foreign key (id_producto) references Productos(id_producto), foreign key (id_pedido) references Pedidos(id_pedido));


CREATE TABLE OrdenesTrabajo(id_ordenTrabajo number(32), fecha date, estado varchar(64), primary key(id_ordenTrabajo));


CREATE TABLE ProductosOrdenTrabajo(id_ordenTrabajo number(32), id_producto number(32), id_pedido number(32), unidades_recoger number(10), unidades_recogidas number(10), primary key(id_ordenTrabajo, id_producto,id_pedido), foreign key (id_producto, id_pedido) references ProductosPedido(id_producto, id_pedido), foreign key(id_ordenTrabajo) references OrdenesTrabajo(id_ordenTrabajo));


CREATE TABLE Incidencias(id_incidencia number(32), id_ordenTrabajo number(32), fecha date, causa varchar(64), primary key(id_incidencia), foreign key (id_ordenTrabajo) references OrdenesTrabajo(id_ordenTrabajo));


CREATE TABLE Paquetes(id_paquete number(32), destinatario varchar(255), direccion varchar (255), primary key (id_paquete));


CREATE TABLE ProductosPaquete(id_producto number(32), id_pedido number (32), id_paquete number(32), unidades_Producto number(10), primary key(id_producto, id_pedido, id_paquete), foreign key(id_paquete) references Paquetes(id_paquete), foreign key(id_producto, id_pedido) references ProductosPedido(id_producto, id_pedido));
